<script src="https://cdn.socket.io/4.5.4/socket.io.min.js"></script>
<script>
    const socket = io();
    
    function toggleReaction(postId, reactionType) {
        fetch(`/react/${postId}`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
            },
            body: JSON.stringify({ reaction_type: reactionType })
        })
        .then(response => response.json())
        .then(data => {
            if (data.success) {
                document.getElementById(`likes-${postId}`).textContent = `👍 ${data.likes_count}`;
                document.getElementById(`hearts-${postId}`).textContent = `❤️ ${data.hearts_count}`;
                document.getElementById(`sads-${postId}`).textContent = `😢 ${data.sads_count}`;
            }
        });
    }
    
    function submitComment(postId) {
        const input = document.getElementById(`comment-input-${postId}`);
        const content = input.value.trim();
        
        if (content) {
            socket.emit('new_comment', {
                post_id: postId,
                content: content
            });
            input.value = '';
        }
    }
    
    socket.on('receive_comment', function(data) {
        const commentDiv = document.getElementById(`comments-${data.post_id}`);
        if (commentDiv) {
            const comment = document.createElement('div');
            comment.className = 'comment';
            comment.innerHTML = `
                <span class="comment-author">${data.username}</span>
                <span class="comment-timestamp" style="color: #999; font-size: 0.85rem;">${data.timestamp}</span>
                <p>${data.content}</p>
            `;
            commentDiv.appendChild(comment);
        }
    });
</script>
